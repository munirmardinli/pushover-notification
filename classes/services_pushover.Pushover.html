    Example

    const pushover = new Pushover({
      token: 'your_api_token',
      user: 'your_user_key',
      debug: true
    });

    pushover.send({
      message: 'Server backup completed',
      title: 'Backup Status',
      sound: 'cashregister'
    }, (err, result) => {
      if (err) console.error(err);
      else console.log('Notification sent:', result);
    });
